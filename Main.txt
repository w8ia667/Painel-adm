# --- Configuração do Youtube-DL (yt-dlp) ---
youtube_dl.utils.bug_reports_message = lambda: ''

ytdl_format_options = {
    'format': 'bestaudio/best',
    'outtmpl': '%(extractor)s-%(id)s-%(title)s.%(ext)s',
    'restrictfilenames': True,
    'noplaylist': True,
    'nocheckcertificate': True,
    'ignoreerrors': False,
    'logtostderr': False,
    'quiet': True,
    'no_warnings': True,
    'default_search': 'auto',
    'source_address': '0.0.0.0' # Isso ajuda em alguns hosts
}

ffmpeg_options = {
    'options': '-vn' # Desativa o vídeo, só queremos o áudio
}

ytdl = youtube_dl.YoutubeDL(ytdl_format_options)

class YTDLSource(disnake.PCMVolumeTransformer):
    def __init__(self, source, *, data, volume=0.5):
        super().__init__(source, volume)
        self.data = data
        self.title = data.get('title')
        self.url = data.get('url') # URL real da música
        self.webpage_url = data.get('webpage_url') # URL do YouTube

    @classmethod
    async def from_url(cls, url, *, loop=None, stream=False):
        loop = loop or asyncio.get_event_loop()
        data = await loop.run_in_executor(None, lambda: ytdl.extract_info(url, download=not stream))

        if 'entries' in data:
            # Pega a primeira entrada se for uma playlist ou resultado de busca
            data = data['entries'][0]

        filename = data['url'] if stream else ytdl.prepare_filename(data)
        return cls(disnake.FFmpegPCMAudio(filename, **ffmpeg_options), data=data)

@bot.event

    print(f'Logado como {bot.user} (ID: {bot.user.id})')
    print('------')

@bot.command(name='join', help='Faz o bot entrar no seu canal de voz atual.')
async def join(ctx):
    if not ctx.author.voice:
        return await ctx.send('Você não está em um canal de voz!')

    channel = ctx.author.voice.channel
    if ctx.voice_client:
        await ctx.voice_client.move_to(channel)
    else:
        await channel.connect()
    await ctx.send(f'Entrei no canal: `{channel.name}`')

@bot.command(name='play', help='Toca uma música do YouTube. Ex: !play Never Gonna Give You Up')
async def play(ctx, *, query):
    if not ctx.voice_client:
        # Se o bot não estiver em um canal de voz, tenta entrar no do autor
        if ctx.author.voice:
            await ctx.author.voice.channel.connect()
        else:
            return await ctx.send('O bot não está em um canal de voz e você também não está!')

    # Se já estiver tocando, parar antes de tocar a nova
    if ctx.voice_client.is_playing():
        ctx.voice_client.stop()

    await ctx.send(f'Procurando e carregando: `{query}`. Isso pode levar um momento...')

    try:
        player = await YTDLSource.from_url(query, loop=bot.loop, stream=True)
        ctx.voice_client.play(player, after=lambda e: print(f'Player error: {e}') if e else None)
        await ctx.send(f'Tocando agora: **{player.title}** ({player.webpage_url})')
    except Exception as e:
        await ctx.send(f'Ocorreu um erro ao tocar a música: `{e}`')
        print(f'Erro ao tocar música: {e}')

@bot.command(name='volume', help='Define o volume do bot (0-100). Ex: !volume 50')
async def volume(ctx, volume: int):
    if ctx.voice_client is None:
        return await ctx.send("Não estou em um canal de voz.")

    ctx.voice_client.source.volume = volume / 100
    await ctx.send(f"Volume alterado para {volume}%")

@bot.command(name='stop', help='Para a música e desconecta o bot do canal de voz.')
async def stop(ctx):
    if ctx.voice_client:
        await ctx.voice_client.disconnect()
        await ctx.send('Desconectado do canal de voz.')
    else:
        await ctx.send('Não estou em um canal de voz.')

@bot.command(name='pause', help='Pausa a música atual.')
async def pause(ctx):
    if ctx.voice_client and ctx.voice_client.is_playing():
        ctx.voice_client.pause()
        await ctx.send("Música pausada.")
    else:
        await ctx.send("Nenhuma música tocando para pausar.")

@bot.command(name='resume', help='Resume a música pausada.')
async def resume(ctx):
    if ctx.voice_client and ctx.voice_client.is_paused():
        ctx.voice_client.resume()
        await ctx.send("Música resumida.")
    else:
        await ctx.send("Nenhuma música pausada para resumir.")
